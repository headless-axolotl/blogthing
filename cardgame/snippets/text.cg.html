9push_some_values : proc[b] {
10 // Native procedures are identified by the '@' symbol before their name.

14main : proc[i, b] {
15 // Store the value 1 in the parameter b (identified by the '.' before the
16 // name of the parameter). I.e. use the buffer with index 1 to push values.
17 copy 1 .b;
18 // Call a procedure defined in the card game language.
19 proc push_some_values[.b];
20
21 // In this case resets the value of the parameter i to 0.
22 del .i;
23 loop .i < #len buf:.b {
24 proc @print[buf:.b[.i], **];
25
26 // The operation/block after the '>' symbol is executed after the body
27 // of the loop. This happens even after continue.
28 } > add 1 .i;
29
30 // The #str directive specifies that the handle afterwards refers to a
31 // string constant and not a symbol. The identifier before the ':' character
32 // specifies the dependency pack where the constant is defined.
33 proc @print[#str core:newline];
34
35 copy 3 .i;
36 copy 1 reg0; copy 1 reg1;
37 // Calculate the 10th Fibonacci value. The registers reg0 and reg1 already
38 // contain the 1st and 2nd number.
39 loop .i <= 10 {
40 // Adds the value of reg0 to reg1, stores the result in reg1 and puts
41 // the previous value of reg1 into reg0.
42 add reg0 reg1 > reg0;
43 } > {
44 // Demonstration that there can be a block after the main body of the
45 // loop.
46 add 1 .i;
47 add 2 reg2;
48 }
49
50 // Strings are surrounded by the '*' symbol.
51 proc @print[*10th Fibonacci number:*, reg1, #str core:newline];
52 proc @print[*reg2:*, reg2, #str core:newline];
53
54 // Labelled scopes. Loops can also be labelled.
55 'label {
56 // Stores the result of reg1 % 2 to reg0;
57 mod 2 reg1 reg0;
58 // A silly if operation to demonstrate the generated bytecode. The body
59 // of the if operation (in this case the break operation) does not need
60 // to be a block. The #sym directive specifies that the handle
61 // afterwards refers to a symbol. This is the default.
62 if ![reg0 == 0 | #sym cat < lasagna] & buf:1[0] != ? break 'label;
63 proc @print[*this should not be in the output*, #str core:newline];
